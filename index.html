04. 변수 : 변수의 종류 : 지역변수 + 전역변수 

{
  let x = 100;    //전역변수
  let y = 200;    //전역변수

  function func(){
      let x = 100;        //지역변수
      let z = "javascript";  //지역변수
      x = 200;           //지역변수 100->200
      y = 300;           //전역변수 200->300      

      document.write(x);
      document.write(y);
      document.write(z);
  }
  func();

  document.write(x);
  document.write(y);
  document.write(z);
}

결과 보기

200
300
javascript

100
300
x

17. 객체 : 데이터를 저장(키와 값) : 표현 방법8 : 객체 속에 함수 

{
  const obj = {
    a: 100,
    b: [200, 300],
    c: {x: 400, y: 500},
    d: "javascript",
    e: function(){
        document.write("javascript가 실행되었습니다.");
    },
    f: function(){
        document.write(obj.d + "가 실행되었습니다.");
    },
    g: function(){
        document.write(this.d + "가 실행되었습니다.");
    }
  }

  document.write(obj.a);
  document.write(obj.b);
  document.write(obj.b[0]);
  document.write(obj.b[1]);
  document.write(obj.c.x);
  document.write(obj.c.y);
  obj.e();
  obj.f();
  obj.g();
}
